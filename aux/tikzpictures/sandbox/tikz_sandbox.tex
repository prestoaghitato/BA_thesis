% based on http://www.texample.net/tikz/examples/double-arrows/
% used in tikz_levinson.tex
\tikzstyle{vecArrow}=[
	very thick,
	decoration={
		markings,
		mark=at position 1 with {
			\arrow[thin, scale=3]{open triangle 60}
		}
	},
	double distance=1em,
	shorten >= 1.4em,
	preaction={decorate},
	postaction={
		draw,
		line width=1em,
		white,
		shorten >= 1.3em
	}
]

\begin{document}

\begin{tikzpicture}
	[
		node distance= 2cm and 2cm
	]
	\draw[help lines, dashed] (0,0) grid (6,6);
	
	\node (a) {A};
	\node[right of=a] (b) {C};

	\draw[vecArrow] (a) to node {bla} (b);
\end{tikzpicture}
